clf = RandomForestClassifier(n_estimators=50,criterion='gini',max_depth=5,min_samples_split=2,min_samples_leaf=10,max_features='sqrt',bootstrap=True,
    n_jobs=-1,random_state=0,class_weight='balanced')
y_predict=[]
y_true=[]
for (train_id, test_id) in cv.split(X, y):
    Xtrain=X.iloc[train_id]
    ytrain=y.iloc[train_id]
    Xtest=X.iloc[test_id]
    ytest=y.iloc[test_id]
    clf.fit(Xtrain,ytrain)
    ypred=clf.predict_proba(Xtest)[:,1]
    y_predict=np.concatenate([y_predict,ypred])
    y_true=np.concatenate([y_true,ytest])

classifier = tf.keras.models.Sequential()

# Adding the second hidden layer

# Adding the output layer
classifier.add(Dense(units = 1, kernel_initializer = 'uniform', activation = 'sigmoid'))

# Compiling the ANN
